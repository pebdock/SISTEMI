#!/bin/bash
for file in /usr/include/* ; do
        if [[ -d ${file} && -r ${file} && ${file} -nt /usr/include/stdio.h ]]; then
                echo ${file};
        fi;
done
Con -r verifico che il file sia readable, -d che sia una directory e -nt per verificare che è più nuovo del file definito

for let in {c..g} ; do
        for file in /usr/include/?${let}* ; do
                if [[ ${#file} -lt 18 || ${#file} -gt 23 ]] ; then
                        echo ${file};
                fi;
        done;
done

#!/bin/bash
exec {FD}</usr/include/stdio.h
if(( $? == 0 )); then
        NUM=0
        while read -u ${FD} -N 1 -r A; do
                (( NUM=NUM+1 ))
        done;
        exec {FD}>&-
        echo ${NUM}
fi;
ricordarsi che -u serve per specificare il file della read , -N per dire quanti char leggere e -r per indicare di non interpretare i backslash words \n \t ecc...
